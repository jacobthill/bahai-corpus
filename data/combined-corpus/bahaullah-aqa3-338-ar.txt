  بسم الّذى به انار افق البيان و نطق قلم الرّحمن
يا کمال قد سمعت نداء ربّک اذ ارتفع بالحقّ و شربت من بحر الوصال اذ کان النّيّر الاعظم مشرقاً من افق الزّوراء طوبى لعين رأت و لاذن سمعت و للسان نطق باسمى المحبوب يا کمال قد ذکرت لدى العرش مرّة بعد مرّة و هذه کرّة اخرى ان اشکرو قل لک الحمد يا اله الغيب و الشّهود انّه ذکرک اذ کان فى العراق و يذکرک فى هذا الحصن الممنوع کذلک نطق لسان المظلوم لتشکر محبوب العالم و تتشبّث بذيل اللّه مالک الوجود دع النّاس و ما عند هم عن ورائک لعمرى سيفنى ما تراه و يبقى لک ما نزّل من هذا القلم الّذى خضعت له الاقلام انّ ربّک لهو الحقّ علّام الغيوب قل يا قوم اتّقوا اللّه و لا تتّبعوا الّذين کفروا باللّه و تمسّکوا بالموهوم ان افتحوا ابصار کم و توجّهوا الی الافق الاعلی تاللّه انّه ظهر بالحقّ و اشرق من ذاک الافق المرفوع ايّاکم ان تتّخذوا له شريکاً او شبيهاً ان انصفوا فى امره و لا تکوننّ من الّذين کفروا باللّه ربّ ما کان و ما يکون قل ان اتّحدوا علی امر اللّه و سلطانه و لا تتّبعوا کلّ فاجر مردود قل هذا يوم به وعدتم فى کتب اللّه و هذا امرٌ خضع له کلّ منصف نطق بالعدل و کلّ مؤمن شرب رحيقى المختوم يا کمال تاللّه لو تطّلع علی حزن قلبى و ما ورد علىّ فى هذا السّجن الّذى سمّيناه لدى الورود بالسّجن الاعظم لتنوح لنفسى کذلک يخبرک ربّک العزيز المشهود انّک لا تضطرب عمّا ذکرناه لک ان اصبر ثمّ اصطبر کذلک يأمرک المظلوم کن مستقيماً علی امر اللّه و حبّه علی شأن لا تمنعک جنود العالم کذلک يأمرک الاسم الاعظم من هذا المقام المحمود قل يا قوم زيّنوا انفسکم بطراز العبوديّة للّه الحق تاللّه الحق هذا يومها لو انتم تعلمون انّا نذکر امتى الّتى آمنت باللّه و خضعت لوجهه اذ کان يمشى انّ ربّک لهو الغفور العطوف کبّر من قبلی علی وجهها و بشّرها بذکرى ايّاها و على وجوه ابنائک الّذين آمنوا باللّه المقتدر المهيمن القيّوم البهاء عليک و على اخيک الّذى کان مذکوراً لدى العرش طوبى لک و له بما سمعتما و توجّهتما الی الافق الاعلى کذلک يجزى اللّه عباده الّذين نطقوا بالحقّ و آمنوا به اذ اعرض عنه کلّ فاجر مردود ثمّ کبّر من قبلی علی احبّائى فى هناک ليشربون من کلماتى رحيق عنايتى و يطيرون فى هواء حبّى و يتوجّهون بقلوبهم الی وجهى و افقى و يفرحون بما نزّل لهم من قلمى عليهم ذکرى و ثنائى و رحمتى و غفرانى انّ ربّک لهو المعطى المشفق المکرم العزيز الودود .
