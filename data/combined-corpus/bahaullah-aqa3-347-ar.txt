	بسم اللّه العزيز الرّحمن 
ان ارض برضى اللّه و انّ رضائه فى تلک الايّام لا تحقق الّا برضاء مظهر نفسه الّذى ظهر بالحق بين العالمين و فزغ عنه من فى السّموات و الارض الّا الّذين عصمهم اللّه بفضل من عنده و انّه لهو المقتدر القدير زيّن هيکلک بطراز الايقان بين ملأ الاکوان ثمّ انطق بثناء ربّک الرّحمن بين عباده الغافلين تاللّه من ذاق حلاوة ذکره انّه لا يشهد من فى السّموات الّا کيوم ما خلقهم الّا من تجد فى وجهه نضرة النّعيم دع النّاس فى خوضهم و خذ لوح اللّه بقوّة مبين ايّاک ان يمنعک شىٌ عن ذکر اللّه الاعظم العظيم سوف يفنى الکلّ و يبقى الملک لربّک الفريد و لو کان للدّنيا شأن عند اللّه ما اودعنا انفسنا بين ايدى المشرکين انّا نکون ناظراً الی مقام ما اطّلع به احد الّا اللّه لذا لا نفزع من ضرّ الظّالمين انّک فامش علی اثرى ثمّ انقطع عن المنکرين تبعّد عنهم و تقرّب الی اصفياء اللّه و انّ فى ذلک خير کثير کذلک القيناک قول الحق لتشکر اللّه ربّک و ربّ ابائک الاوّلين و تقول ان الحمد للّه ربّ العالمين .
