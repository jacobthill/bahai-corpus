	الاقدس الامنع الاعلى 
يا ابانور ان استمع نداء الظّهور الّذى ارتفع من شجر الطّور اللّه لا اله الّا هو المقتدر المختار کنت فى السّجن کما کنّا فيه ان اشکر و قل لک الحمد يا الهى بما وفّقتنى بذلک اشهد انک انت العزيز الوهاب ان يا کمال قم علی امر ربّک الغنىّ المتعال على شأن لا تمنعک سبحات الجلال و لا اشارات الّذينهم کفروا بالمبدء و المآل ادع الناس بالحکمة و البيان کذلک امرک ربّک الرحمن فى الزّبر و الالــــــــــــــــــواح کمالی لا تحزن بما ورد  عليک فى سبيلی قد ورد علينا فى سبيل اللّه ما لا حملته الاکوان کنت شريکنا فى المصائب و البلايا يشهد بذلک ربّک مالک الابداع طوبى لک بما وفيت ميثاق اللّه و عهده و کنت طائفاً حول البيت اذ کان ربّ البيت فى ارض العراق انّا کنا معک اذ کنت بين ايدى المشرکين و نريک بما انت عليه انّ ربّک لهو العزيز العلّام ان رأيت المحمود قل اين مهربک لعمرى هذا المقام المحمود الّذى نزّل حکمه فى الفرقان ضع الصّنم و توجّه الی مولی الامم هذا ينبغى لک و لمن فاز بهذا اليوم الّذى جعله اللّه مطلع الايّام يا محمود ان استمع ما ينادى به ربّک الودود من السّدرة النّوراء الّتى ارتفعت علی البقعة الحمراء وراء قلزم الکبرياء اللّه لا اله الّا هو العزيز الغفّار دع کلّ ما سمعت ثم توجه الی آيات ربّک لعمرى لو تجد حلاوتها تدع الورى عن وراک و تتوجّه الی موليک و تقول آمنت بالّذى ارسلک يا ايّها المسجون فى اخرب الدّيار تفکّر فى نفسک اتختار الصّمت فى امر ربک بعد الّذى ارتفع صرير القلم الاعلى بين الارض و السّماء لعمرى لو تنصف تنوح و تبکى فى العشىّ و الاشراق هل تظنّ انّ ايمانک ينفعه او اعراضک يضرّه لا و منزل الآيات انّه يدعوک خالصاً لوجه اللّه اذ احاطته المشرکون من کلّ الجهات دع الحميّة و تمسّک بحبل مالک البريّة هذا خيرٌ لک عمّا خُلق فى الارضين و السّموات لا تقنع بالموهوم توجّه الی ربّک القيّوم الّذى اتى بقدرة و سلطان ان يا کمال ان وجدت فى وجهه نضرة الجمال فَالق عليه آيات ربّک ذوالعظمة و الاجلال و الا فاَعرض عنه ثمّ اقبل بقلبک الی اللّه مالک يوم المعاد انا ذکرناه فى هذا اللّوح بما سبقت رحمة ربّک و الّا لا يتحرّک على مثله قلم اللّه العزيز الجبّار الّا بان يتوب و يرجع الی الّذى به لاحت الوجوه و نادى المناد کبّر من قبلی على وجوه الّذين اقبلوا و مستّهم الشّدائد فى ايّام شداد و بشر هم بعنايات ربّهم ثمّ اتل لهم ما نزّل من لدى اللّه العزيز الغفّار انا نذکر هم فى السّجن و نزلنا لکلّ واحد ما يثبت به اسمائهم فى کتاب اللّه رب الارباب انّما البهاء عليک و علی ولدک و اهلک و علی الّذين استقاموا علی هذا الامر الّذى به اخذت الزّلازل قبائل الآفاق کن بين الناس کما اُمرت به من قبل ليُرى منک ما تقرّبه عيون الابرار اذاً امسکنا القلم و لو انّه اراد ان يجرى علی اسماء احبّتى الی يوم الّذى لا يُحدّ بالاذکار کذلک نزّلنا عليک الآيات فضلاً من لدنّا لتقرئها بين العباد الّذين خرقوا السبحات باسم منزل الآيات و کسّروا صنم الوهم بهذا الذّکر الّذى نزّل من سماء القدرة و الاقتدار.
