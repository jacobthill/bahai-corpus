	هو العليم 
اسمع يا اسمى ثم اسطع عن کثيب الحمرا بسطوع المسک عن رضوان ربّک المنّان العزيز القدير ليعطّر منک الممکنات و يضعنّ روايح الامکان و يقلبنّ الی وجه ربّک الرّحمن العزيز العليم ان يا رحيم فاخرج عن غرف السّکون و کن منادياً من لدنّا بين العالمين فانقطع عن نفسک فى سبيل ربّک ثم هاجر الی اىّ ارض تريد باسمى المريد الغالب العظيم و ان تکون واحداً فاستانس بذکر ربّک المهيمن العزيز الحکيم و انّک لو تسمع ما علمّک جمال العلم تاللّه يوثّر قولک على کلّ شيئ بل على کلّ جبل راسخ منيع قم عن مقامک ثمّ اشتعل بهذه النّار الّتى منها اشتعلت کلّ الاشياء ليحدث منک حرارت حبّ اللّه فى قلوب الموحّدين تاللّه يا رحيم انّا وجدنا ملأ البيان فى وهم الّذى لن يقاس بوهم احد فى الملک لذا بکت عيون سرّى لنفسى الوحيد الغريب قل يا قوم فانصروا الغلام و انّه لهو الّذى به رفع دينکم و علت اسمائکم و ظهرت قدرتکم و لاح جمال اللّه المقتدر العزيز البديع ان يا قوم لا تحرموا انفسکم عن هذا الفضل الّذى ارتفع بالحق خافوا عن اللّه الّذى اليه منقلبکم و مثوى العالمين قم باذن اللّه ثمّ ذکّر النّاس بما يذکرک روح القدس فى هذا اللّيل المنير ان يا اسمى ان کنت فقيراً فى الضّعف فاستغن باسمى القادر و ان کنت عليلاً فى الجسد فاستشف باسمى الشّافى وان کنت کليلاً فى النّطق فاستنطق بسلطانى النّاطق العليم الفصيح و انّک اذا اردت ان تبلّغ النّاس الی اللّه اذاً ينطق الرّوح علی لسانک و تجرى عنک ينابيع الحکمة و البيان و کان اللّه علی ما اقول شهيد کذلک يبشّرک ربّک لتبشّر النّاس برضوان قرب منيع .
