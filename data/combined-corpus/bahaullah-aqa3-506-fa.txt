	هو اللّه تعالى شأنه الحکمة و البيان 
يا نور مالک ظهور در طور عرفان که مقرّ سجن مقصود عالميان است ترا ذکر مى نمايد مقصود آنکه نداى الهى حزب خود را جذب نمايد و بانچه سبب ارتفاع کلمة اللّه است مؤيّد دارد در ليالی و ايّام اوليا را ذکر نموديم و بعفّت و عصمت و امانت امر فرموديم بعضى آگاه گشته و کوثر عرفان را از يد عطاى حق جلّ جلاله نوشيده و برخى انکار نمودند و بر کفران نعمت الهى دهان گشودند عمل نمودند آنچه را که زفرات مقرّبين متصاعد و عبرات مخلصين نازل سبحان اللّه عباد غافل بما عند هم مشغول و از ما عند اللّه غافل و محجوب حزب بيان ظالمترين احزاب مشاهده شدند ويل لهم و سحقاً لهم انکروا آيات اللّه بعد انزالها کفروا بنعمته بعد اظهار ها قل خافوا اللّه و لا تدحضوا الامر بما عند کم سوف ترجعون و ترون انفسکم فى اسفل الجحيم نامه شما که بعبد حاضر ارسال نمودى در ساحت اقدس عرض و بشرف اصغا فائز طوبى لک يا نور انت الّذى فزت بايّامی و کنت ساکناً فى ظلّى انظر ثمّ اذکر اذ کنت قائماً لدى الوجه و فزت بآيات اللّه ربّ العالمين طوبى لک و لابيک الّذى آمن فى يوم فيه ارتعدت فرائص الاسماء فى خشية اللّه العزيز الحميد نشهد انّه فاز بالمقام الاعلى و الرّتبة العليا نسأل اللّه تعالى ان ينزل عليه فى کلّ حين ما ينبغى لسماء رحمته و بحر جوده انّه فضّال کريم ذکر اسامى چند در نامه ات بود هر يک بآثار قلم اعلى فائز ليشکروا ربّهم الغفور الکريم هذا ما نزّل لجناب ميرزا صادق عليه بهاءاللّه هو السّامع المجيب ذکر من لدنّا لمن فاز بانوار الوجه و آمن باللّه الفرد الخبير يا صادق انّا نذکرک بالصّدق و نوصيک بما يرتفع به امر اللّه مالک هذا اليوم البديع قل الهى الهى قد اجذبنى ندائک الاحلى اسألک بآياتک الکبرى و نفحات وحيک يا مولی الورى بان تؤيّدنى على ما ينبغى لايّامک ثمّ اجعلنى فى کلّ الاحوال متمسّکاً بک و بآياتک انّک انت المقتدر علی ما تشاء لا اله الّا انت القوىّ القدير و نذکر من سمّى بمحمّد الّذى اقبل الی اللّه العزيز الحکيم يا محمّد بحکمت الهى تمسّک نما و بفضلش تشبّث هر روز منادى عزّت عباد را خبر داده و ميدهد و آنچه اليوم مقبول و محبوبست تبليغ امر الهى است طوبى از براى نفسى که بروح و ريحان اهل امکان را بمقصود عالميان هدايت نمود و نذکر الجواد الّذى فاز بالافق الاعلى و نوصيه بما ينبغى لايّام اللّه ربّ العالمين و نذکر من سمّى بمحمّد قبل باقر و نبشّره بعناية اللّه ربّه انّا ذکرنا هما فى الواح شتّى لا يعادل بحرف منها کنوز الارض يشهد بذلک کلّ عارف بصير و نذکر من سمّى بمحمّد قبل اسمعيل ليفرح بذکر ربّه و يکون من الشّاکرين يا عبد الرّحيم اسمع النّداء من شطر عکّا انه ذکرک بما انجذبت به افئدة الملأ الاعلى و قرّت عيون المقرّبين يا محمود انّ المظلوم فى السّجن الاعظم و يذکر اوليائه بما لا ينقطع عرفه بدوام ملکوت اللّه العزيز الحميد و نذکر من سمى بمحمّد قبل علی و نوصيه بالاستقامة الاکبرى فى هذا الامر الذى به ارتعدت فرائص الاسماء کذلک نطق لسان العظمة اذ کان مستوياً على عرش البيان فى هذا المقام الرّفيع يا عبد الکريم قد توجّه اليک وجه القدم و يوصيک بالامانة و الدّيانة و بما يرتفع به امر اللّه العليم الخبير و نذکر محمّداً قبل حسين بذکر به تنجذب الافئدة و القلوب نسأل اللّه ان يجعلک ثابتاً راسخاً فى هذا الامر الاعظم و هذا النّبأ العظيم يا عبداللّه طوبى لاذن سمعت ندائى و لعين رأت آثار قلمى و لوجه توجّه الی انوار وجهى المنير يا درويشعلی حضر اسمک لدى المظلوم اقبل اليک وجه القدم من شطر سجنه الاعظم و يذکرک بذکر تزيّنت به کتب العالم طوبى لمن شهد و عرف و ويل للغافلين و نذکر فى هذا الحين من سمّى بعبد الکريم و نلقى عليه من آيات ربّه ليکون من الرّاسخين علی هذا الامر الّذى به زلّت الاقدام و اضطربت افئدة المريبين يا غلامرضا مالک اسما در اين حين ترا ذکر مى نمايد و جميع دوستان را و کلّ را وصيّت ميفرمايد بآنچه که سبب اعزاز کلمه الهى است ما بين خلق يا حزب اللّه در الواح کلّ را باتّحاد و اتّفاق امر نموديم چه که اتّفاق سبب اعلاء کلمه بوده و هست و نصرت امر بجنود اعمال و اخلاق و بتقوى اللّه ربّ العالمين يا اهل البهاء بآنچه از قلم اعلى جارى شد تمسّک نمائيد ان شاء اللّه بعنايت حق جلّ جلاله کلّ موفق شده و ميشويد بآنچه رضاى او در آنست يا نور جميع منتسبين را از لسان مظلوم تکبير برسان بشّر هم بعنايتى و ذکّر هم بآياتى منتسبين اسم کمال لازال در نظر بوده و هستند قبل از صعود و بعد از صعود اسم کمال مخصوص او از قلم اعلى نازل شد آنچه که شبه و مثل نداشته انّا طهّرناه حين صعوده و غفرناه فضلاً من عند نا و قرّبناه الينا و ادخلناه فى الفردوس الاعلى رحمةً من لدنّا انّ ربّک هو الرّحمن الرّحيم الحمد للّه ربّ العالمين .
