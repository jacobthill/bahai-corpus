	هو المنادى بين الارض و السّماء 
يا ايّها النّاظر الی الافق الاعلی ان استمع نداء مالک الورى انّه يجذبک الی مقام لا تخوّفک سطوة العالم و لا تمنعک جنود الامم الّذين غفلوا عن هذا الامر الاعظم العظيم سيفنى ما تراه اليوم و يبقى الملک و الملکوت للّه الواحد المقتدر العليم الحکيم قل قد حکم ابوجهل الزّمان علی محمّد رسول اللّه و السّامرىّ علی الکليم و اليهود علی روحى العزيز المنير و قام اهل الفرقان علی منزل البيان و الرّقشاء قامت علی اللّه المهيمن العزيز القدير لانّها لدغت اصفيائى الّذين طاروا فى هوائى و توجّهوا الی وجهى و شربوا رحيق محبّتى و تمسّکوا بحبلی و تشبّثوا بذيلی المطهرّ المقدّس العزيز المنيع فانظر فى النّاس و اعمالهم انهم اتّخذوا الرّقشاء لانفسهم اماماً من دون اللّه و ارتکبوا ما ناح به الملأ الاعلى و الّذين طافوا حول عرش العظيم قد اخذ ها اللّه بسلطان من عنده و قدرة من لدنه و لکنّ النّاس اکثر هم من الغافلين انّهم فازوا بما اردا و انفقوا ارواحهم و اموالهم فــــــــــــــــــــــــــــــــــــــــــى سبيل اللّه ربّ العالمين و لکنّ القوم خسروا بما عملوا سوف يرون انفسهم فى خسران مبين قد نزّل من القلم الاعلى فى ذکر هم ما لا تعادله کنوز الارض کذلک قضى الامر من لدن ربّک الکريم طوبى لقوىّ ما اضعفته جنود الظلم و لناطق تمسّک باسمى الحکيم قل تمسّکوا بحبل الحکمة فى کلّ الاحوال هذا ما حکم به اللّه فى اکثر الالواح انّه لهو المبيّن العليم البهاء عليک و على من کان مستقيماً فى هذا الامر الّذى به اضطرب کلّ جاهل مريب.
