	بسم اللّه الاقدس الابهى
هذا کتاب من لدنّا الی الّذى اقبل الی الوجه فى يوم فيه اضطربت الارکان لينقطع عمّا عند النّاس و يتمسّک بهذا الحبل الّذى به تحرّک الامکان ان استمع نداء اللّه من هذا الشّطر المقدّس الابهى علی البقعة النّوراء من السّدرة الخضراء انّه لا اله الاّ انا المقتدر العزيز الوهاب قم علی ذکر اللّه بين خلقه و ذکّر النّاس بهذا النّباء الّذى منه ناح الطّاغوت و اخذت الزّلازل من فى البلاد قل يا قوم قد اتى القيّوم و ليس لاحد من مناص الاّ بان يتوجّه الی مطلع الآيات الّذى ينطق بالحق انّه لا اله الاّ هو العزيز المختار قل يا قوم انّ الغيب ينطق بهذا اللّسان بينکم ان اعرفوا و لا تتّبعوا کلّ مشرک مرتاب من لدى الغيب ظهر هيکل المشهود و من جمال المشهود ثبت حکم الغيب کلّ ما يظهر منه انّه يظهر من عنده اذاً اشرقت شمس التّوحيد من افق التّجريد طوبى لاولی الابصار قل يا قوم انّه لم يزل کان و لا يزال يکون بمثل ما قد کان من انکره قد انکر الّذى ينطق بالحقّ فى قطب الاکوان بانّه لا اله الاّ هو العزيز المنّان ضع الّذين کفروا و ما عندهم لعمرى لايتکلّمون الاّ باهوائهم و لم يريدوا الاّ بان يدخلوا الرّيب فى قلوب الّذين کسّروا الاصنام انّ الّذى شرب رحيق المعانى من هذه الکأس البيضاء انّه لا تمنعه الاشارات عن التّقرّب الی اللّه منزل الآيات ما کان همّهم الاّ بان يجدوا ما يعترضوا به علی اللّه مالک يوم المآب قل کلّ ما ينزل انّ المنزل هو اللّه يشهد بذلک کلّ الذرّات من انکرنى فقد انکر الغيب و يلعنه کلّ الاشياء و عن ورائها لسان العظمة و الاقتدار انّک اذا فزت بلوح اللّه قل لک الحمد يا الهى بما ذکرتنى بعد الّذى کنت بين ايدى الفجّار.
