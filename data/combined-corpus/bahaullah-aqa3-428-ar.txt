	هو الاقدس الاعظم 
يا نور ان استمع ما يدعوک به مالک الظّهور انّه يجذبک و يعلّمک و يعرّفک و يبلّغک الی مقام لا ترى فى الملک الّا مالک الملکوت الّذى يحبيک بعد ما حضر کتابک فى هذا المقام المحمود قد قرئناه و وجدنا منه عرف خلوصک لِلّه ربّ ما کان و ما يکون انّا نذکرک و اخاک الّذى تشبّث بذيل اللّه مالک الغيب و الشّهود و نذکر من اتى بعده انّ ربّک لهو العطوف الغفور انّا نکبّر احبّائى فى هناک و نذکر هم بذکر قلمى الاعلى الذى ينطق فى کلّ الاحيان انّه لا اله الّا انا الذّاکر و المذکور.
