	الاقدس الاعظم 
ان يا هيکل انّک بنيت بامر ربّک ولکنّ النّاس لا ينظرون و منهم من ينتظر ان ينظرک طلباً لهواه الا انّه من الغافلين قل انّى خلقت بارادة ربّى و زيّنت بطراز الاسماء فى يوم الّذى فيه طرّز اللّه الملکوت باسمه الابهى من انتظر بعد ذلک انّه من الهائمين قل بظهورى ارتفع ذکر اللّه و به خلقت هياکل لا يعلم عدّتها احد الاّ اللّه العزيز العليم يا ملاء‌الرّوح ان اقبلوا بالرّوح و الرّيحان الی هيکل الرّحمن هذا ينبغى لکم ان انتم من العارفين ايّاکم ان يمنعکم عباد الّذين يعبدون الاوهام و لا يکوننّ من الشّاعرين قل ضعوا الاوهام قد اتى العلاّم علی ظلل الغمام و الموحّدون ينظرون و يقولون تبارک القيّوم الّذى اتى بسلطان مبين قل لا تلتفتوا الی الدّنيا و ما فيها ضعوها عن ورائکم مقبلاً الی موليکم القديم قل دعوا الکنائس قد ظهرت کنيسة اللّه علی هيکل الانسان ان اقبلوا اليه ثم اذکروه بهذا الذّکر الحکيم فيه ارتفع نداء النّاقوس باسمه القدّوس و انتم فى حجبات انفسکم لمن الميّتين قوموا من مراقد الهوى مقبلاً الی اللّه مالک الاسماء هذا ما وعدتم به فى الالواح لو انتم من النّاظرين اجيبوا داعى اللّه بينکم ثم اخرجوا من البيوت مقبلاً الی الملکوت کذلک ينصحکم ربّکم الصّادق الامين انّ الّذين اعتکفوا فى الکنائس هل خلقناهم للاکل و القعود لا و نفسى العزيز الودود بل لهذا اليوم الّذى فيه ظهر المکنون باسمه المهيمن علی العالمين قل ان احضروا بين يدى الرّب و لا تمنعوا انفسکم عمّا خلقتم له هذا من فضلی عليکم ان انتم من المتفرّسين
