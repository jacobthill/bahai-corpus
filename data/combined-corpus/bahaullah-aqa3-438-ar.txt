	هو الاقدس( لوح هيکل) 
ان يا هيکل ان افرح فى نفسک بما بينناک من زبر البيان و عمّرناک باسم ربّک الرّحمن و قدّسناک عن الظّنون و الاوهام انّ ربّک لهو العليم الخبير انّک تنطق فى کلّ الاحيان يا ملأ الاکوان انّ الهيکل الّذى بنی من الطّين خيرٌ ام الّذى خلق من ايادى رحمة ربکم الرّحمن الرّحيم قل انّى هيکل اللّه بينکم ايّاکم ان تترکونى مقبلين الی اوهامکم اتّقوا اللّه و لا تتّبعوا کلّ متوهّم بعيد قد خلقنى اللّه باسمه و جعلنى مزيّناً بطراز ذکره بين ملأ الاکوان تعالی الرّحمن الّذى جعلنى مصدر امره و مطلع وحيه بين العالمين قل ما اراد اللّه فى الکتاب من هذا الاسم الّا هيکلی لو انتم من العارفين قد رفعنى اللّه بسلطانه و جعلنى عَلَم الهداية بين عباده ان اسرعوا الىّ و لا تکوننّ من المتوقّفين قل بقيامى قام امر اللّه فى ما سواه و بظهورى تمّ ما نزّل فى الالواح و کمل ما ذکر فى التّورية و الانجيل ايّاکم ان تخرّبوا هيکل اللّه ان عرفوا ما اخبرناکم به من قبل ثمّ انصروا ربّکم العزيز الحکيم انّ هذا لبرّ اللّه لو انتم من الشّاعرين قل انّ هذا لکلمة اللّه لو انتم من العالمين ان يا ملأ الرّوح ايّاکم ان تنکروا ما ثبت به ذکر اللّه بينکم و ايّاکم ان تتّبعوا المحتجبين طوبى لميّت حىّ بروح اللّه و تحرّک من النّسمة الّتى مرّت من قميص الفضل و الاحسان و ويلٌ للمنکرين لا تتّبعوا الّذين يدّعون الايمان بالّذى ارسلنا من قبل فلمّا ظهر ما هو المقصود کفروا به و تمسّکوا بما عند هم الا انّهم فى خسران مبين قل انّ کلمة اللّه تنطق و تمشى لو انتم من الفقهين اذا تمشى تمرّ منها نسمات السّبحان علی من فى الاکوان و تتضوّع نفحات الرّحمن فى الامکان طوبى لمن وضع ما سويه و اقبل و وجد و قال لک الحمد يا من فى قبضتک جبروت الابداع و ملکوت الاختراع اشهد انّک انت المقتدر على ما تشاء لا اله الّا انت العزيز الکريم بک نوّرت الوجوه و ظهرت السّاعة و غنّت الورقا و تحقّق ما ذکر فى کتب اللّه فى ازل الازال کذلک قضى الامر و لکنّ النّاس فى حجاب عظيم قل اتوجّهتم الی الهياکل الّتى صنعت بايادى عبادنا و وضعتم هيکل اللّه عن ورائکم ان هذا الّا ظلم مبين اتظنّون تعبدون اللّه ربّکم بل اوهامکم لو انتم من المتفرّسين لو کنتم تعبدونه انّه ظهر بالحق لم انکرتموه فأتوا ببرهانکم ان انتم من العارفين طوبى لمن نبذ الغدير و سرع الی الکوثر الّذى جرى من اصبع القدير الا انّه من الفائزين من اقبل الی هذا الوجه فى اقلّ من آن لعمرى انّه هو المقبول عند المحبوب و الّذى بعد انّه من اهل السّعير قد ارتفع النّداء علی شأن سمعه کلّ سمع و انتم غفلتم و کنتم من المبعدين ضعوا ما عند کم ثمّ استمعوا صرير هذا القلم الّذى به جرى الکوثر و السّلسبيل اذا ينطق تنطق الاشياء انّه لا اله الّا هو المقتدر القدير قل انّى لهو الّذى باسمه نزّلت الالواح و زيّنت صحائف اللّه ربّکم ان انتم من المتفرّسين و تمنعون انفسکم من لقاء ربّکم ان افتحوا الابصار و لا تعقّبوا کلّ مشرک اثيم انّه فى البقعة النّوراء و الارض البيضاء يدع الکلّ الی اللّه الفرد الخبير و حمل الشّدائد و البلايا على شأن ما حمله الابن من قبل و لا من النّبيين و المرسلين قد تغيّر طراز التّقديس من مفتريات الّذين کفروا بالبيّنات و اذا تتلى عليهم الآيات قالوا ان هذا الّا سحر عظيم قالوا بمثل ما قالوا الا انّهم من الاخسرين طوبى لوجه تنوّر بانوار الوجه و لعين توجّهت الی هذا المنظر الکريم .
