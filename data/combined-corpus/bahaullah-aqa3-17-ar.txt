	الاقدم الاعظم 
ذکر من لدّنا لمن ذکر مولاه ليجذبه نداء ربّه الی مقام قدّر فى الالواح قد انار الوجه من افق ارادة ربّک و غنّ عندليب البقاء فى ملکوت الانشاء و اضاء المصباح قد نطق کلّ شئ بثناء ربّک و الّذين غفلوا اولئک من الاموات هذا يوم فيه نصب الصّراط و وضع الميزان و الامر للّه العزيز المنّان ثمّ اعلم قد حضر کتابک لدى الوجه و نزّل لک ما اضائت به الآفاق طوبى للسان نطق بذکر الرّحمن و لقلب اقبل و لوجه توجّه الی العزيز الوهّاب قم علی خدمة اللّه و ذکره بقيام تستقيم به الاقدام قل ان اجتمعوا علی الشّريعة الّتى شرعناها بالحق لينکسر ظهر الّذين کفروا باللّه مالک يوم المعاد ان اغتنموا الايّام ثمّ اجتمعوا علی ما اراد اللّه مالک الانام انّک انت يا عبد ان انصر ربّک بالبيان علی شأن تفرح به نفحات القميص بين الامکان انشاء اللّه لم يزل و لا يزال بثناى حق ناطق باشى و بر اتّحاد قلوب ساعى اين بسى واضحست که اختلاف سبب افتراقست و اتّحاد و وفاق علّت ظهور قدرت و اقتدار کمر خدمت محکم کن و ناس را به‌اسم اعظم منقطعاً عن الکلّ دلالت نما اين است حکم محکم که از افق قلم قدم اشراق فرمود تمسّک به منقطعاً عن العالمين انّما البهاء عليک و علی الّذين آمنوا باللّه الفرد الواحد العليم الخبير.
