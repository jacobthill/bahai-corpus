	هو الاقدس الاعظم العلىّ الابهى 
شهد اللّه انه لا اله الّا هو و الّذى ينطق انه لمکلّم موسى فى الطّور و النّاطق فى المعراج بالسّر المستور و المقصود الّذى اليه صعد الرّوح و انه لکتاب مسطور و لوح اللّه المحفوظ و الکنز المخزون و السّر المکنون و الرّمز الّذى ما اطّلع به الکاف و النون هذه کلمة جعلها اللّه رحمةً و نوراً و بهآءً من عنده علی من توجه الی بيته المعمور و ناراً و قهراً لمن کفر باللّه المهيمن القيّوم الحمد للّه الّذى اظهر ما هو الموعود و انزل من سماء الجود ما اهتزّ به اهل القبور منهم من قام و سرع الی بحر الحيوان و منهم من اعرض و ادبر و اشرک باللّه الواحد العزيز الغفور و الحمد للّه الّذى نطق بمظهر امره و مشرق وحيه و دعا الکلّ الی لقائه و کوثر وصاله و جعل لقائه ذات لقائه ليتّم القول فيما انزله علی حبيبه اللّه الّذى رفع السّموات بغير عمد ترونها و سخّر الشمس و القمر کل يجرى لاجل مسمّى يدبّر الامر يفصّل الآيات لعلّ النّاس بلقاء ربّهم يوقنون و ليتّم ما ذکر من قبل فتوقّعوا ظهور مکلّم موسى من الشّجرة علی الطّور طوبى لمن وجد عرف بيان اللّه و اقبل اليه و ويلٌ لمن اعرض و کان من اصحاب القبور و بعد قد تشرّف الخادم بلقاء من فاز بالمنظر الاظهر و سمّى بعلی قبل اکبر عليه بهاء اللّه مالک القدر الّذى توجّه و اقبل الی ان دخل البقعة النّورآء المقام الذى تنطق فيه سدرة المنتهى لا اله الّا هو المقتدر المتعالی العزيز المحبوب و سمعت منه ذکر کم و اقبالکم و ارتقائکم و توجّهکم و ثنائکم و قيامکم و خلوصکم للّه موليکم و استقامتکم علی امره الّذى به اضطرب کل قوىّ و جهل کلّ عالم و انصعق کلّ عارف و تبلبل کلّ سابح و تقهقر کل مقبل و اراد ان اکتب لکم ما يدل لخلوص ايّاکم فى اللّه لذا اطلقت زمام قلمى على قدرى و مسکنتى لعل اللّه يحدث به ما يفرح به قلبکم و قلوب الّذين اقبلوا الی اللّه و انقطعوا عما سواه و لو ان لمثلی ما ينبغى ان اذکر ذرّة فى الملک فکيف الّذين تزيّنوا بطراز التّوحيد و اقرّوا باللّه الواحد المقتدر العليم الحکيم و لکنّ العبد فى کلّ الاحوال اکون متشبّثاً بذيل کرمه و متمسّکاً بحبل الطافه بان يقبل منى و من عباده ما اعمل و يعملون فى سبيله و اذکر و يذکرونه بين خلقه انّه لهو المقتدر العطوف الرّحيم ان شاء اللّه لم يزل و لا يزال باشراقات انوار شمس حقيقت منوّر باشيد و بافق ظهور الهى ناظر و متوجّه اليوم آنچه اعظم اعمال بوده و هست بعد از عرفان حق استقامت است بايد آنجناب که از کوثر بيان آشاميده اند جميعرا باين مقام بلند اعلى بخوانند ناس ضعيفند و متذکّر لازم دارند ان شاء اللّه بايد بکمال همّت در ارتفاع کلمه مبارکه جهد نمائيد تا از نار محبّة اللّه جميع اهل آن ديار مشتعل گردند و بکمال اتّحاد و اتّفاق بذکر سلطان حقيقى مشغول شوند و قليل اين عبد را کثير شمرند چه که اين خادم فانى ليلاً و نهاراً بتحرير آيات الهى مشغول است بشأنى که مجال نمييابد تا آقايان و دوستان را بعرايض خود زحمت دهد از جميع اميد عفو بوده و هست و جناب ملّا علی اکبر عليه من کل بهاء ابهاه تفصيل را ذکر ميفرمايند استدعا آنکه احبّاى الهى که در آن ارض بنار محبّة اللّه مشتعلند و بشطر اللّه ناظر از جانب اين عبد تکبير برسانيد و بگوئيد امروز روز ذکر و ثناست و امروز روز خدمت و طاعت است چه که هر عملی که  اليوم خالصاً لوجه اللّه ظاهر شود از براى او مثل و شبهى نبوده يشهد بذلک ما انزله الرّحمن فى الالواح البهاء عليکم و على اخيکم و على من معکم من لدى اللّه العزيز الوهّاب خادم ١٤ جمادى الاولی سنه ١٢٩٣
