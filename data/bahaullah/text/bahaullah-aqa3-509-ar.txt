	هو اللّه الامنع الاقدس الابهى
فسبحان الّذى نزّل الآيات بالحق من جبروت عزّ علياً و ينطق بالحق فى ملکوته الاعلی بلسانه الابدع الاحلی لعلّ الناس يتّخدنّ الی جهة الرّوح تلقاء عرش ربّهم الرّحمن بالحق سبيلاً قل يا قوم اتّقوا اللّه و لا تتّبعوا انفسکم ان ارتقبوا فضل اللّه و انّ فضله کان عليکم محيطاً و ان استشرقت عليکم شمس الکلمات عن افق مليک الاسماء و الصّفات اذاً لا تستکبروا ثمّ اسجدوا لجمال ربکم الّذى کان فى جبروت البقاء باسم البهاء فى ملکوت الانشاء بالعلىّ مذکوراً ان يا قوم لا تکفروا بآيات اللّه بعد انزالها و لا تتّخذوا الشّيطان لانفسکم وليّاٌ ايّاکم ان لا تضطربوا حين الّذى تضطرب فيه الاشياء و تنفطر فيه السّماء و تنشقّ ارض السّفلى و تندکّ الجبال و تتموّج البحار و تضع کلّ ذات حمل حملها و يأخذ السّکر سکّان السّموات و الارض و کذلک نزّل الامر بالحق من قلم القدس و کان اللّه على ما اقول شهيداً ضعوا ما يحجبنّکم عن اللّه ثمّ اصعدوا بجناحين التّقديس الی نداء السّماء التى قد ارتفعت بالحق و کانت على اسمى الابهى بالسّر مرفوعاً تاللّه قد هبّت نسايم الفضل عن مشرق العدل و بها حملت کلّ الاشياء ان انتم بذلک خبيرا و سوف يضع الامکان حمله اذاً تشهد المشرکين يفرّون من اليمين و الشّمال و لن يجدنّ لانفسهم مقراً امينا و کذلک انبئناک من نبأ الامر لتطّلع باسرار الّتى کانت خلف سرادق العزّ مقنوعاً قل يا قوم خذوا کأس الحمرا من انامل البها ثم انقطعوا عن کلّ من فى الارض و السّماء و ان استطعتم فى انفسکم فارکبوا باسمى الاعلى علی فلک الحمرا و سيروا على بحر الکبريا لتصلنّ الی مقعد الّذى کان عن رمى المشرکين محفوظا ان اسمعوا يا قوم نداء اللّه من کلّ الجهات و لا تلتفتوا الی الّذينهم کفروا بالّذى آمنوا به و کانوا عن سبل الحق بعيداً و من المشرکين من کان مجاهداً بامواله و نفسه لاعلاء امرى فلمّا ظهر عليه جمالی اذاً کفر به و کان علی عقبيه منقلبا و انّک انت يا عبد و لو ما حضر منک کتاب بين يدى العرش ولکن لمّا ذکر اسمک انزلنا اليک ما تقرّ به عيناک و عيون الّذينهم آمنوا باللّه و کانوا علی صراط العزّ مستقيماً ان استقم فى نفسک لئلّا يزلّک شيئ عن الصّراط ثمّ اشکر ربّک بما نزل عليک لوح عزّ بديعاً کذلک مننّا عليک من بدايع فضلنا و ارسلنا اليک ما تجد به روايح ربّک و تتّخذ فى ظلّ عصمة ربّک مقاماً کان علی الحقّ اميناً و من النّاس من اعرض عن اللّه جهرةً و اذا تتلی عليه آيات اللّه يسودّ وجهه و ينقلب الی اهله مبغوضاً و منهم من قال بانّها ما نزّلت علی الفطرة قل تاللّه انّ الفطرة قد خلقت بحرف منها و يشهد بذلک ما يجرى من قلم قدس بديعاً و منهم من افترى علی اللّه و قال ان هذا الّا ساحر يسحر النّاس و بما خرج من فمه حبطت اعماله و انکر النبيّين و المرسلين و کان فى دين اللّه بغياً قل يا قوم خافوا عن اللّه و لا تقولوا ما قاله المشرکون حين الّذى اشرقت شمس القدم عن مشرق الحجاز بسلطان مبيناً و کذلک قال المشرکون فى زمن کلّ نبىّ الی ان انتهت الايّام الی ايّام اللّه و اشرق جمال القدم عن افق اسم عليّاً و انّک انت يا عبد قدّس اذن القدس عن کلمات المشرکين ثمّ ادخل فى غمرات هذا البحر باذن ربّک الرّحمن لتجد لئالی حکمة ثميناً قل انّا اخذنا قبضة من التّراب و عجنّاه بمياه الامر من لدنّا و نفخنا فيه روحاً من امرنا ثمّ زيّناه باسمائنا الحسنى فى ملکوت الانشا و ارفعناه الی مقام الّذى اشتهر اسمه بين کلّ صغير و کبيراً فلمّا بلغ اشدّه و اطمئنّ فى نفسه اذاً استکبر علی نفس اللّه و سلطانه الی ان حارب معه جهرةً و کذلک کان حکم القضا علی جمال ربّک العّلی الاعلی من اصبع اللّه مقضياً قل يا قوم اتّقوا اللّه و لا تفرطوا فى جنب اللّه تاللّه هذا من نبأ الذى به قرّت عيون اهل ملأ البقا و کان خلف حجبات النّور بعصمة اللّه محفوظاً فلمّا تمّت الميقات اشرق عن افق القدس ببرهان الّذى کان علی العالمين محيطاً و انتم ان تنکروا هذا البرهان فباىّ برهان يثبت ايمانکم لا فو الّذى اشرقت السّموات بنور وجهه اذاً لن تجدنّ لانفسکم الی الحقّ دليلاً قل يا قوم اذا دخل عليکم رسول اللّه بامره قوموا عن مقاعد کم و خذوه بانامل التّسليم ثم استنشقوه ان وجدتم منه رايحة اللّه محبوبکم اذاً لا تنکروه کذلک نزّل الامر من جبروت القدس تنزيلاً من لدن عزيز قديراً و يا قوم ان اتّبعوا ملة اللّه و دينه و لا ترتکبوا ما نهيتم عنه فى الکتاب اتّقوا اللّه و کونوا على الامر مستقيماً فسوف يدعون المشرکون احداً من هناک و يدخلون فى قلبه بغض اللّه و مظهر نفسه و يرسلونه الی الدّيار و اليکم و معه ما يقلّب النّاس عن صراط عزّ منيراً فطوبى لمن لن يتحرّک من اهتزاز الشّرک و يکون مستقيماً علی امر مولاه تاللّه انّه لخير عند ربّک عمّن خلق بين السّموات و الارض جميعا ذلک من انباء الغيب نوحيها اليک فضلا ً من لدنّا عليک و علی کلّ موقن بصيراً و البهاء عليک و علی الّذين آووا فى ظلّ هذه السّدرة الّتى ارتفعت بالحق و کانت فى قطب الرّضوان بيد اللّه علی الفضل مغروساً .
