	بسم اللّه العليم الخبير 
هذا کتاب من لدنّا الی الّذى استضاء من انوار الّتى اشرقت من افق ارادة ربّه العلىّ الاعلی فى ايّام کشف جمال الغيب عن وجهه الغطاء و ظهر بسلطان اسمه الابهى بين ملأ الإنشاء و ينطق فى کلّ شىٌ انّه لا اله الّا هو العزيز الکريم طوبى لک بما خرقت الاحجاب و ما منعک السّبحات عن التّوجّه الی ربّک العزيز الوهّاب و احيتک روح اللّه و ايقظتک نسمات التّى تضوّعت نفحاتها بين العالمين طوبى لک بما عاشرت مع الّذى اصطفيناه لذکرنا و عزّزته حين الّذى ورد عليک بذکر اللّه محبوب العارفين خذ لوح اللّه و اثره ثمّ دع اللدّينهم کفروا و لو يأتيکم بصحف الاوّلين و الاخرين لانّ اليوم لا ينفع احداً شىٌ الّا من تمسّک بهذا الحبل الّذى علّق بين السّموات و الارضين انّ الّذينهم فى مرية فى ايّام ربّهم اولئک ليس لهم نصيب من هذا الکوثر الّذى جرى من فم عناية ربّک الرّحمن فى قطب الجنان فسوف ياخذ هم اللّه بقهر من عنده و انّه اشدّ الآخذين فسوف يأتيکم من يدعو کم الی الشّيطان کذلک اخبر کم قلم الرّحمن من قبل فى لوح عزٍ بديع قل يا قوم تجنّبوا عنه و لا تتّبعوا الّذين کفروا بالّذى آمنوا تاللّه ان هم الّا فى ضلال مبين قل يا قوم اتعبدون الاسما و تکفرون بموجدها فويل لکم من عذاب يوم عظيم و قد قدّر لک فى سماء الامر مقام عزٍ کريم انّه يؤيّد الّذين استقاموا علی امره و انّه هوموّضی اجور المحسنين و قد غشت الاوهام کلّ الانام الّا من تشبّث بذيل ربّک المقتدر العزيز العلّام انّه من اعلی الخلق يصلّينّ عليه اهل ملأ الاعلی ثمّ ملائکة المقرّبين قل عنده علم کلّ شىء لا يعزِب عن علمه ما هو المستور فى صدور الخلايق اجمعين يجزى کلّ نفس بما کسبت و انّه بکلّ شىٌ عليم نسأل اللّه بان يظهر منک ما تقرّبه عيون الموحّدين و يجعلک علم الهداية بين البريّة و قائد العباد فى هذا اليوم الّذى شقّت فيه سبحات الجلال و اتى اللّه بجمال مبين و فى کلّ حين يؤيّدک علی ذکره و ثنائه بين عباده و انّه لعلی کلّ شىءِ قدير ان افرح بعناية اللّه و رحمته و انّ رحمته احاطتک بالفضل و لا يعقلها الّا الّذينهم طاروا فى هذا الهواء الّذى قدّسه اللّه عن ظنون المريبين و لو يکشف الغطاء عمّا قدّر لک فى هذا اللّوح لتتحيّرنّ العباد بل ينصعقنّ لذا سترناه فضلاً من لدنّا و انّه لهو الغفور الرّحيم نسأل اللّه بان لا يعقّبه البدا و ينزل عليک فى کلّ حين ما يجذبک اليه و ينقطعک عمّا سواه لا اله الّا هو المعطى الباذل العليم الحکيم و النّفحة الّتى فاحت عن قميص ربّک العلىّ الاعلی عليک و علی من معک من عباد اللّه المخلصين.
