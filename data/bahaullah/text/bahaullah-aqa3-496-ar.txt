	هو المنادى باسمه القيّوم 
ذکرٌ من لدنّا لمن اقبل الی الوجه اذ ظهر من هذا الافق الّذى تزيّن بذکره کتب اللّه المهيمن القيّوم تاللّه قد ظهر ما کان مکنوناً فى حجب الغيب و اتى من کان مستوراً فى علم اللّه العزيز الودود هذا يوم بشّر به محمّد رسول اللّه من قبل و من قبله الانجيل و التّوراة و الزّبور قل انّه اتى بالحق و اظهر صراطه و انزل آياته و انطق الاشياء علی انّه لا اله الّا هو الحقّ علّام الغيوب ان افرح بما ذکرناک فى السّجن الاعظم و ارسلنا اليک هذا اللّوح المسطور خذ کتاب اللّه بقوة من عنده ثمّ اعمل بما امرت به من لدى اللّه مالک الوجود فاسئل اللّه بان يظهر منک ما ينبغى لظهوره و استوائه و سلطانه الّذى احاط الغيب و الشّهود طوبى للسان نطق بذکر اللّه و لقلب اشتعل بنار حبّه و لبيت ارتفع فيه اسمه العزيز المحبوب کذلک نوّرنا افق اللّوح بنيّر البيان طوبى لمن فازو ويل لکلّ غافل محجوب البهاء المشرق من افق البقاء عليک و علی الّذين نبذوا ما عند هم ورائهم مقبلين الــــــــــــــــــــــــی اسمنا القيّوم الّذى اذ ظهر ظهر الفرح الاکبر و الفزع الاعظم و نطق العندليب قد فتح باب السّماء و اتى مالک الاسماء بسلطان مشهود .
