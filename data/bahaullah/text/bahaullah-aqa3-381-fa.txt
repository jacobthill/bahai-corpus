	بسم ربّنا الاقدس الاعظم العلىّ الابهى 
حمد و ثنا افنان سدره الهيّه را لايق و سزاست که جنود ظنون و صفوف اوهام ايشانرا از نبأ عظيم و صراط مستقيم منع ننمود بفضل اللّه از کوثر يفعل مايشاء آشاميدند و از تسنيم يحکم ما يريد نوشيدند سطوة فراعنه ظلم و شوکت جبابره اعتساف ايشانرا از مطلع عنايت و الطاف محروم نساخت باصبع توکّل حجبات را دريدند و بعضد توسّل اصنام را فرو ريختند ابصارشان بافق اعلى ناظر و وجوهشان بانوار وجه متوجّه ارياح مختلفهء عالم ايشان را از سکون و اطمينان باز نداشت و اضطراب و تزلزل امم را در قلوبشان راه نه جلّ من خلقهم و عرّفهم و قرّبهم و عزّ من رفعهم و فضّلهم و انطقهم و اراهم و اسمعهم سبحانک اللّهم يا الهى و مقصودى و محبوبى و محبوب العالم و معبودى و معبود الأمم اسألک باسمک الّذى به ظهرت الساعة و و اشراطها و القيمة و ظهوراتها بان تنزل من سماء عنايتک علی افنانک ما ينبغى لعزّک و قوّتک و جودک و کرمک و اقتدارک اى ربّ تراه فى کلّ الاحوال قائماً علی خدمة امرک و ناطقاً بثنائک هو الّذى اقبل اليک فى يوم فيه اعرض عنک من فى ارضک و اعترض عليک من فى بلادک و انفق ما عنده لاعلاء کلمتک و اظهار امرک و انتشار آياتک و عزّتک يا الهى لو يريد الخادم ان يذکر ما ظهر منه فى ايّامک ليرى قلمه عاجزاً و لسانة قاصراً اسألک يا مقصود افئدة المقرّبين و محبوب قلوب المخلصين بآياتک الکبرى و نفسک العلىّ الابهى بان تظهر من افق العناية و الالطاف اعماله فى ايّامک ليرى الکلّ ظهورها و انوار ها کما يرون الشمس و اشراقها و انوارها استغفرک يا الهى من ذکرى هذا المقام الّذى جعلته فوق مقامى اسألک بنور توحيدک الّذى منع عنه عبادک بان تويّده فى کلّ الاحيان علی ما کان انّک انت المقتدر المستعان لا اله الّا انت العليم الحکيم روحى و نفسى و ما عندى لحبّکم الفدا و لقيامکم الفدا قد کان الخادم قاعداً صامتاً متفکّراً فى آيات اللّه و ظهورات عظمته و قدرته و سلطانه اذاً ورد الافنان عليه بهاء اللّه الابهى بکتاب حضرتکم اخذت و فتحت کانّى وجدت من کلّ حرف من حروفاته عرف محبّة اللّه مقصودنا و مقصود کم و مقصود من فى السّموات و الارض فلمّا قرئت و عرفت اخذنى اهتزاز حبّکم علی شأن طارت باجنحة الاشتياق الی انَ حضرت امام وجه اللّه مالک التّلاق و عرضت ما فى الکتاب و هذا ما نطق به لسان العظمة فى الجواب قوله عزّ بيانه و جلّ برهانه هو السّامع المجيب يا افنانى عليک بهائى و عنايتى و رحمتى الّتى سبقت من فى السّموات و الارضين قد حضر العبد الحاضر و قرء ما نطق به لسانک فى ذکر اللّه و ثنائه و قوّته و قدرته و اقتداره و عزّه و عظمته و سلطانه طوبى للسانک بما نطق بالحقّ و لقلمک بما جرى منه ما يجد کلّ ذى دراية عرف الخضوع و الخشوع و الانقطاع فى حبّ اللّه مالک الابداع يشهد قلمى الاعلى بانّک تنوّرت بنور التّوحيد و شربت من قدح التّجريد اذ کان النّاس فى شرک مبين لعمرى ما فاز احد بيوم اللّه و ما ظهر فيه الّا من نبذ العالم و تمسّک بالاسم الاعظم الّذى به زلّت اقدام العلماء و نسفت جبال العلوم و ارتعدت فرائص الاسماء و اضطربت افئدة العرفاء و ناح کلّ ظالم بعيد يا افنانى اشکوا اليک من الّذين اعرضوا عن الّذى اتيهم بآيات لا تعادلها کتب العالم لولاه ما ظهر ما کان مکنوناً فى علم اللّه ربّ العالمين الی حين يوم اللّه را ادراک ننمودهاند و بقطره از بحر کلمه مبارکه يفعل ما يشاء فائز نگشته‌اند چه اگر فى الحقيقه بنور اين کلمه عليا فائز ميشدند باب اعتراضات را مسدود مينمودند و بجان بشطر محبوب امکان توجّه ميکردند لعمر اللّه آذان موجوده در عالم لايق اصغاء اين نداء نبوده و نيست و همچنين ابصار قابل مشاهده افق اعلی نه اين يوم يوميست عظيم و اين امر اعظم من کلّ عظيم للّه الحمد آنجناب مکرّر فائز گشتند و از بحر بيان رحمن آشاميدند قد سبقت ما دونک فى خدمة اللّه و نصرة امره يشهد بذلک من عنده امّ الکتاب از حق جلّ جلاله ميطلبيم عباد خود را حفظ فرمايد از اوهامات نفوس غافله ظالمه محيله بايد کمال جهد را در اين فقره مبذول داريد و از حق از براى کلّ حفظ طلب نمائيد انّه معک و يسمع ما اردت من بدايع فضله و الطافه انّه هو السّامع المجيب در جميع احوال بايد عباد را متذکّر نمود که شايد از بحر آگاهى بياشامند و باشراقات انوار دانائى فائز شوند اگر موافق حکمت باشد از قبل مظلوم اولياء حق را تکبير برسانيد و بما يرتفع به امر اللّه امر نمائيد شايد اشياء فانيه ايشان را از نعماء باقيه الهى محروم نسازد افنان سدره کل لدى الوجه مذکورند و بعنايات مخصوصه فائز اوراق را تکبير برسانيد و بعنايت حق بشارت دهيد البهاء المشرق من افق سماء فضلی و رحمتى عليک و علی من معک و يحبّک و يسمع قولک فى هذا النّبأ العظيم انتهى للّه الحمد امواج بحر عنايت ظاهر و انوار شمس شفقت مشهود حينى که اين خادم فانى تلقاء وجه حاضر و دستخط آنحضرت را عرض نمود آفتاب فضل بشأنى مشرق و لائح که اين عبد و امثال او از ذکرش عاجز و قاصر و در احيان عرض مناجات مکرّر اين کلمه عليا از لسان مولی الورى ظاهر نعم ما نطق به الافنان فى يوم اللّه مالک الاديان آيات منزله خود شاهد و گواهست احتياج بدونش نبوده و نيست بفضل اللّه آن حضرت مزيّنند بآنچه که سزاوار يوم اللّه است جناب افنان آقا سيّد علی عليه بهاءاللّه الابهى تشريف آوردند و آنچه آنحضرت فرمودند ذکر نمودند و تلقاء عرش بشرف اصغاء فائز و همچنين فقرات مذکوره در دستخط للّه الحمد کلّ بشرف اصغاء فائز و بانوار آفتاب قبول مزيّن و منوّر اين خادم فانى خدمت حضرت افنان عليهم بهاءاللّه الابهى و من کلّ نور انوره و من کلّ جمال اجمله و من کلّ عناية افخرها و من کلّ بهجة اظهرها و من کلّ نعمة اسبغها و من کلّ مائدة اتّمها عرض سلام و ثنا معروض ميدارد از حق تعالی شأنه سائل و آمل که عباد خود را مؤيّد فرمايد و بما ينبغى موفّق دارد تا کلّ قصد مقصد اقصى و افق اعلى نمايند و از فيوضات ايّام محروم نمانند انّه علی کلّ شىٌ قدير الثّناء و البهاء و الذّکر و التکبير و السّلام علی حضرتکم و علی من
