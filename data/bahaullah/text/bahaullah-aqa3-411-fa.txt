	المحبوب 
هذا کوثر الحيوان قد جرى من قلم الرّحمن طوبى لمن شرب منه و کان من الفائزين قل بهذا الماء يحيى الارواح ان اشربوا يا قوم و لا تلتفتوا الی اليمين و الشّمال کذلک يأمرکم قلم الاعلی فضلاً من لدى اللّه العلىّ الاعلى ان اعملوا ما امرتم به من لدن عليم حکيم و الّذى ما شرب انّه ميت لا يصدق عليه حکم الحيوة هذا تنزيل من اللّه و انّه لهو المنزل القديم روحى لحبّکم الفدا و لخدمتکم الفدا بعد از زيارت کتاب آن حضرت که فى الحقيقه مطلع سرور و مشرق ابتهاج بود لدى العرش حاضر و جميع تلقاء وجه عرض شد و فى الحقيقه نفحات محبوبه از آن نامه مبارکه در هبوب بود ديگر احتياج دليل و برهان نيست نفس آيات منزله از سماء مشيّت الهيّه شاهد و گواهست بر فضل اکبر و عنايت کبرى لذا بايد در کلّ احيان بکمال فرح و سرور محبوب امکان را ستايش نمائيد اين عبد که از نزول آيات منيعه بزبان کليل عوض آن حضرت نيستى و فنا و عدم استحقاق اين عباد عطاياى لا تحصيه الهيّه را اظهار نمودم و بصد هزار لسان شکر مينمايم محبوب عالم را چه که کمال الطاف نسبت بآن حضرت ظاهر فرمود انّ الخادم يشکره بلسان کلّ الاشياء و ذرّات العالم و رمول البيداء و بصد هزار عجز و نياز استدعا مينمايم که لدى اللّه مقبول واقع شود بارى فضل حق نسبت بآن جناب زياده از ذکر اين فانى بوده و خواهد بود بهر يک از اسامى که در نامه آنحضرت بود واحداً بعد واحد آيات بديعه منيعه از سماء عنايت رحمانيّه نازل و ارسال شد ان شاء اللّه بصاحبانش برسد تا از سلسبيل وحى بنوشند و از کوثر بيان رحمن قسمت برند هينئاً لهم و از براى بعضى از نفوس مذکوره الواح الهيّه متتابعاً مترادفاً رفته و مخصوص جناب ملّا علی جان يک لوح بديع منيع از سماء بيان نازل و از قبل ارسال شد ولکن عجبست از بعضى از نفوس که در اخذ الواح مخصوص احبّاى الهى بسيار ساعى و جاهدند و بعد از اخذ بعضى ميرسانند و بعضى نگاه ميدارند از حق ميطلبيم کل را بديانت و امانت موّفق فرمايد فآهٍ آهٍ ثمّ آهٍ آهٍ عين الخادم تبکى بما ورد علی هيکل الامانة بين البريّة لو کنت مأذوناً لذکرت ما ينوح به الملأ العالم ولکن استر و اصبر انّ ربّى لهو السّتّار الصّبّار الغفور الکريم و عرض ديگر آنکه جواب عريضه مانکجى صاحب از سماء مشيّت نازل و ارسال شد و چون بلسان پارسى نازل شده شايد در بعضى اثر نمايد و ثمر آن ظاهر شود اين عبد از روى آن سوادى بخطّ نسخ نوشته و ارسال داشت که هر يک از دوستان بخواهند از روى آن سواد بردارند تا اصل لوح بى نقص و عيب بصاحبش برسد بارى اگر سواد آن نوشته شود و تصحيح شود و ببعضى داده شود شايد ثمر نمايد ولکن در جميع احوال بمقتضاى حکمت عمل فرمائيد قسم بعرف قميص که اگر جميع ناس مستعدّ بودند همين يک لوح کفايت مينمود و اينکه مرقوم داشته بوديد که جهرةً حرکت فرمائيد و يا در ستر باشيد تلقاء عرش عرض شد فرمودند ستر بهتر است و آسوده تريد عجبست که هنوز مردم مطّلع نشده اندکه مطلع وحى الهى و مشرق امر ربّانى مقصودش فساد و نزاع و جدال نبوده و نيست اى کاش عرف اراده حق را اقلّ من سمّ الابره اداراک مينمودند در اين صورت يقين مينمودند که آنچه اراده فرموده اعزّ از کبريت احمر و اعلى عمّا خلق فى الارض بوده چون بوئى از گلزار ارادهء حق نبرده‌اند اين است که خوفاً لانفسهم متعرّض اولياى حق شده و ميشوند در جميع امور و احوال ناظر بحکمت باشيد و شايسته نيست نسبت بدولت حال احدى بکلمه نا‌لايقه تکلّم نمايد فى الحقيقه با اعراض علما و قساوتى که آن نفوس را اخذ نموده حضرت سلطان بسيار خوب سلوک فرموده در هر صورت دولت رعايت اهل ملّت را مينمايد چه که ارتکاب امرى که سبب ضوضاء عامّه خلق شود نزد دولت مقبول نبوده و نيست چه که بايد حفظ تخت و بخت خود نمايد اين قدر معلوم آنجناب بوده که در منظر اکبر ذکر سلطان به نيکوئى شده اگر يکى اين از علماى مظلوم سلطان ميشد آنوقت قدر همه سلاطين نزد شما معلوم و واضح ميگشت کاش شيخ حسين ظالم و مظلوم شيراز را ميديديد يکى اسمش مظلوم که صد هزار ظالم پناه برده و ميبرد بخدا از شرّ او و حلم او کذلک يکلّمک ربّک و يمزح لتفرح و يفرح کلّ عبد بصير انتهى و اينکه در باره محمّد تقى نوشته بوديد بلی چند يومى بساحت اقدس وارد و مشرّف شد و در هنگام رجوع مخصوص باو فرمودند که متوجّه باش تا شياطين ترا از علّيين بسجّين نبرند يعنى مضمون کلام مبارک اين بود و حال اين عبد اميدوار است که راجع شود اگر کسى او را ديد بگويد تو بنده خدا و من بنده خدا با کمال فقر و نيستى ميگويم که قسم بآفتاب تقديس که بر تو مشتبه شد و فريب خوردى و از تو ميطلبم که رجوع نمائى و چندى در آن ارض با آن نفوس معاشرت کنى ولکن با کمال بصيرت باشى تا بر تو معلوم شود آنچه بر تو مجهولست اليوم آن مرتبه رفتى ولکن ملتفت نشدى اين عبد دوازده سنه با او بوده و بامر حق بخدمت او مشغول يا ليت کنت معى و عرفت ما عرفته و اين چند کلمه للّه ذکر شد که شايد رحيق اصفى را تبديل ننمائى و بکأس غبرا معاوضه نکنى و الامر بيد اللّه ليسقى و يمنع انّه لهو المقتدر علی ما يشاء و اينکه مرقوم فرموده بوديد اگر عرايض قبل جواب دارد ارسال شود آنچه از سماء عنايت نازل قبل و بعد و حين همه را کفايت خواهد نمود و اين که مرقوم فرموده بوديد که الواح ناريه از براى بعضى آورده بود اگر همان الواح را صاحبانشان در آن تفکّر نمايند شهادت بر غفلت کاتب ميدهند و ملتفت ميشوند و ذکر علي عسکر و شيرعلی و ميرزا که در آخر کتاب آنحضرت بود عرض شد و نسبت بهر يک عنايات الهيّه ظاهر و مشرق نسأل اللّه بان يؤيّد هم و يوفّقهم و يرزقهم خير ما عنده انه لهو المعطى الکريم حسب الامر آنکه در ستر امور و ارسال پوسته بسيار جهد نمائيد چه اگر شياطين داخل و خارج مطّلع شوند که از ساحت اقدس اخبار ميرسد البتّه فساد نمايند و شايد بعضى اظهار خدمت بخيالشان خطور نمايد و ببعضى از امرا و وزرا اخبار دهند که شايد مقرّب شوند در هر حال ستر لازم و واجبست و تأخير جواب عرايض را سبب کلّى اين بوده و خواهد بود انتهى اگر قسمى شود اين فقره مستور ماند هميشه الواح منيعه بديعه مخصوص دوستان ارسال شود عرض ديگر آنکه خدمت دوستان الهى  يعنى نفوسى که بافق اعلى ناظرند و از رحيق استقامت آشاميدهاند هر يک را از قبل اين خادم فانى تکبير ابدع اعلى برسانيد ابن جناب ع ط عليه و علی ابيه بهاء اللّه که ذکرش در کتاب آنحضرت بود تکبير لا يحصى از اين عبد معلّق بالطاف آنحضرت است و همچنين جناب کربلائى مهدى ساير دوستان عليهم بهاء الرّحمن و البهاء الابهى علی حضرتکم و على من معکم اهل سرادق عصمت ربّانيّه بابدع اذکار و اعليها و ابهاها آنحضرت را ذاکر و بتکبير امنع اقدس مکبّر و همچنين دوستان ارض سجن کلاً و طرّاً عرض خلوص خدمت آن حضرت معروض ميدارند خادم فى
