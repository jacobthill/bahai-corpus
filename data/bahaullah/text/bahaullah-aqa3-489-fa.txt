	هو المشفق العطوف الغفور الرّحيم 
تاللّه يا اسمى الجيم قد قرء لدى العرش کتابک و وجدت منه نفحات حبّک للّه يشهد بذلک قلمى الاعلی فى هذا اللّيل الّذى فيه ينطق جمال القدم بما تضوّع منه عرف الحيوة بين العالمين و انّک اذا فزت به قم و قل لک الحمد يا اله العالمين ان يا اسمى الميم ان استمع النّداء من شطر السّجن لعمرى انّ لسانى يشهد بحبّک للّه و باقبالک اليه ان اطمئن بفضل ربّک و قل لک الثّناء يا محبوب العالمين ان يا اسمى الالف تشهد الف الابداع بانّک انت الّذى سمعت النّداء و اقبلت الی الافق الاعلى و تمسّکت بالمعروف و استقمت على حبّ مولاک اذ کان مضطرباً کلّ قوىّ امين ان يا اسمى اللّام انّ ربّک العلّام يناديک من مقرّ عرشه العظيم و يشهد لک بما تقرّ به عينک و يسرّبه قلبک و يأخذ عرف العناية کلّ صغير و کبير لا تحزن من شيئ انّه يسمع و يرى و ينزل لک ما تفرح به قلوب العارفين لعمرى لو تجد نفحات هذه الايام لتطير من الاشتياق و تصيح فى البيت لک الفضل يا من فى قبضتک ملکوت السّموات و الارضين انّا خلقناک و ربّيناک و اسمعناک و اريناک منظرى الکريم يا اسمى الجمال عليک بهاء اللّه فى کلّ حين و بعد حين و قبل حين انت الّذى تمسّکت بالحبل الاعظم و حملت الرّزايا فى حبّ اللّه ربّ العالمين انّا کنّا معک اذ دخلت فى السّجن و کانت معک انفس معدودات انّ ربّک لهو البصير الخبير ان اذکر من سمّى بعلی قبل اکبر انّه ممّن و فى بميثاق اللّه و عهده و استشهد فى سبيلی المستقيم انّا نذکره فى هذا الحين و نقول عليک بهاء اللّه و بهاء من فى الملأ الاعلی و بهاء الّذين فازوا بهذا الامر البديع يا علی قبل اکبر انّک انت دخلت الرّمس و محبوب العالم يذکرک فى هذا المقرّ الرفيع انت الّذى توجّهت الی وجه موليک و قطعت البرّ و البحر الی ان دخلت شاطى البحر الّذى ينادى کل قطره منه قد تمّ الميقات و اتى الرّحمن بسلطان مبين و اقبلت الی افق الظّهور و قمت لدى الباب و فزت بانوار الوجه و سمعت نداء اللّه المهيمن العزيز الحميد و کنت فى جوار رحمته ايّاما معدودات ثم رجعت باذنه و دخلت السّجن فى حبّه انّ ربّک لهو الشّاهد السّميع طوبى لمن يذکره بعده بما نطق به القلم الاعلی فى هذا المقام المنيع نشهد انّه ممّن انفق روحه فى حبّ مولاه فى يوم فيه زلّت اقدام العارفين انّا نذکره و الّذين معک ليشهد الکلّ بفضل اللّه و رحمته انّه لهو المعطى الغفور الکريم ثم نبشّرک بفضل اخر انّ ربّک لهو المبشّر الخبير انا قد غفرنا اخاک الّذى صعد الی اللّه و نشهد انّه ممّن اقبل الی الافق الاعلى و بلغ الغاية القصوى کلّ ذلک من فضلی عليک لتکون من الشاکرين انّه فى الافق الابهى يشهد بذلک لسان عظمتى فى هذا المقرّ المنير انّا نذکر فى هذا المقام من سمّى بعلی قبل نقى ليبقى ذکره بدوام اسم اللّه الملک المقتدر العزيز الحميد ان يا قلم الاعلى ان اذکره بالرّوح و الرّيحان ثمّ اشهد له بما شهد له الرّحمن انّ ربّک لهو المعلّم الامين قل طوبى لک يا من صعدت الی الرفيق الاعلى و توجّهت الی المقام الاسنى اشهد انّک قد سمعت النّداء و اقبلت و آمنت و کنت من الفائزين انت الّذى ما منعتک شئونات الخلق عن الحقّ قمت علی الذّکر و الثّناء بين ملأ الانشاء و توجّهت بوجهک الی وجه اللّه المشرق من هذا الافق المنير طوبى للّذين فازوا بهذا المقام و نعيماً  لکلّ مقبل اقبل الی اللّه العزيز الجميل ثمّ نذکر من سمّى بعندليب الّذى طار فى هواء محبّة الرّحمن و فاز بظهور اللّه فى يومه البديع انّا نذکره باحسن الذّکر و نرسل اليه نفحات الآيات من هذا المقام الّذى فيه ينطق لسان العظمة الملک للّه العزيز المنيع يخاطبه جمال القدم و يقول عليک ثناء اللّه يا من کنت ناطقاً بذکر ربّک و عليک بهاء اللّه يا من کنت ناظراً الی مشرق فضل ربّک العليم کذلک ذکرنا الّذين سمعوا نداء اللّه و اقبلوا اليه بقلوبهم و شهدوا بما شهد الرّحمن اذ استوى علی عرشه العظيم نعيماً لهم و طوبى لهم بما فازوا فى هذا الحين بکوثر ذکر ربّهم المقتدر القدير هل تعادل بهذا الفضل کنوز العالم لا و اسمى الاعظم و لکنّ النّاس اکثر هم من الرّاقدين بشارة بعد بشارة بما توجّه وجه القدم من شطر سجنه الاعظم الی من سمّى بمحمّد قبل علی الّذى فاز بعرفان اللّه مالک الايجاد انّا نذکره بلحن اللّه ربّ الآخرة الاولى بذکر تمرّ به رائحة القميص بين العالم و يتضوّع عرف الرّحمن فى الامکان انت الّذى اقبلت الی قبلة الافاق و آمنت بالّذى اعرض عنه اکثر العباد طوبى لک بما فزت برحيق الحيوان الّذى ادارته انامل عطاء ربّک الرّحمن بين الامکان اشهد انّک تقرّبت و توجّهت و عرفت و اخذت و شربت باسم اللّه مالک الاديان هل من ذى اذن ليسمع لحن اللّه و هل من ذى بصر ليشاهد انوار الوجه هل من ذى شمّ يجد رائحة الفرح الّذى علّق علی ظهر اللّه المقتدر العزيز المنّان انت فى الرّفيق الاعلى و ربّک الابهى يراک و يذکرک ليکون ذکره اية لمن فى الابداع يا جمال قد سمعنا منک ما کان شاهداً لخضوعک للّه و خشوعک لوجهه و عجزک و ابتهالک لدى اللّه المقتدر العزيز الوهّاب انّه يکون معک فى کلّ الاحيان و يذکرک و الّذين معک انه لهو العزيز البصّار و نکبّر من هذا المقام الاعلی و المقرّ الاسنى علی اهلک و من نسب اليک انّ ربّک لهو المبيّن المختار لا يعزب عن علمه من شيئ يذکر من يشاء بما يبقى به ذکره بدوام ملک اللّه مالک الانام اى جمال کتابت بلحاظ مالک اسماء فائز و جواب آن از سماء مشيّت الهى نازل بشأنى که هر کلمه آن شهادت ميدهد بر فضل محبوب عالم و عنايت او ان اشهد و کن من الشّاکرين و مخصوص هر يک از اسماء که در کتاب آنجناب بود آيات بديعه منيعه نازل اى جمال قسم بشمس محبّت که از افق عالم مشرق است و باعلی الضيّا و ابهى النّور ظاهر و باهر که اگر جميع ناس در ساعتى از ساعات بقلوب مقدّسه مطهّره توجّه نمايند و نداء اللّه را اصغا کنند و اطوار اوراق سدره منتهى را مشاهده نمايند البتّه کلّ از فيض بحر اعظم محروم نمانند و در اين فجر منير بعنايت حق فائز گردند ولکن انّهم فى غفلة و ربّک لهو الشّاهد الخبير اهل خانه و منتسبين شما جميع لدى العرش مذکورند و جميع بعنايت الهى فائز شده و ميشوند لا تحزن من شيئ قد قدّر لک ما تفرح به القلوب انّ ربّک لهو الصّادق المخبر الامين يا اسمى الجمال قد توجّه فى هذا الحين وجه ربّک الی من سمّى ببزرگ ليجد نفحات الوحى و ينطق بثناء ربّه الخبير انّ الّذى توجّه الى اللّه انّه يتوجّه اليه من عنده و انا العليم من نطق بهذا الاسم الاعظم يوقن بانّه کان مذکوراً لدى العرش يشهد بذلک ربّک و انا الشّهيد کبّر من قبلی على وجهه قل تاللّه قد فُزتَ بالفوز الاعظم اذ ذکرک مالک القدم فى هذا اللّوح الحفيظ ان افرح بفضل موليک ثمّ اشکره انّ ربّک لهو السّميع هذا يوم فيه انجذبت الاشياء من نداء مالک الاسماء و يسمع من کلّ ذرة من الذّرات تهلّل و تکبّر و تتحرّک شوقاً الی ظهور اللّه فى هذا المقام المبين يا جمال ان اشهد ثمّ انظر ثمّ اذکر ما رأيت بعينک اذ کنت قائماً لدى الباب و کان متوجّهاً اليک وجه اللّه ربّ العالمين ان افرح بفضلی ثمّ عنايتى ثمّ مواهبى و رحمتى الّتى سبقت الاشياء و بحر کرمى الّذى احاط العالمين و ما ذکرت فى الّذين اقبلوا الی المظلوم بشّر هــــــــــــــــــــم بذکرى ايّاهم ليکوننّ من الفرحين قد عرض لدى العرش العبد الحاضر کلّ اسم کان مذکوراً فى کتابک و نزّلنا له ما فاح به عرف الفضل بين السّموات و الارضين طوبى لهم بما فازوا بعرفان اللّه فى ايّامه و تمسّکوا بحبله المنير يا اهل الطّاء لعمر اللّه انّ المقصود يذکر کم و يناديکم من هذا المقام البعيد و يدعو کم الی مقام لا ياخذه الفناء انّ ربّکم العليم لهو الشّاهد الامين قد ذکرنا کم مرةً بعد مرّةً ان افرحوا بفضل ربّکم و کونوا من الشّاکرين ان ابشروا  بما توجّه اليکم وجه اللّه من هذا الافق البديع قولوا لک الحمد يا اله العالم و مالک القدم بما تحرّک باسمنا قلمک الاعلى و تضوّع منه عرف عنايتک لهؤلاء العباد نشهد انّک انت الفضّال و نحن من السّائلين و ما ذکرت فى اهل الهاء و الميم انّا نزّلنا لهم الآيات فضلاً من لدنّا و انا الکريم و نزّلنا لملّة الکليم فى هناک ما اردته من فضل ربّک الرّحيم و نذکر اهل الشّين و السّين و الميم الّذين فازوا برحيق البقاء الّذى فکّ ختمه باصبع ارادة مالک الانشاء طوبى لهم ثمّ طوبى لهم و حسن مآب ان يا احبّائى فى الشّين تاللّه قد ذکر کم مالک الاسماء بآيات لا يراها المحو فى ممالک الابداع انّ مالک الاختراع يشهد بذلک و عن ورائه کلّ عارف علّام يا رضا قد سمعت النّداء مرّهً بعد مرّةٍ ان استمع فى هذا الکرّة الاخرى و توجّه بالوجه الاطهر الی المنظر الاکبر و بالقلب الانور الی افق ظهور ربّک العزيز الوهّاب قل لک الثناء يا مالک البقاء و لک الذّکر يا من بيدک زمام الايجاد اشهد انّک قد قرّبتنى و شرّفتنى و عرّفتنى و اسمعتنى ندائک الاحلی فى ملکوت الانشاء و انّک انت الکريم الفضّال اسألک بالاسم الاعظم بان تجعلنى ناطقاً بذکرک و تؤيّدنى على الاستقامة الکبرى على امرک الّذى به انقلب الاسماء و ناحت قبائل الارض کلّها الّا من شاء کرمک الذى احاط الامکان و نذکر الّذين فهيناک ليحرّکهم عرف آيات ربّهم الرّحمن و نکبّر من هذا المقام علی وجوههم و نوصيهم بما ينبغى لهذه الايّام ان يا قلمى الاعلى ان اذکر من سمّى بطالب ليفرح بذکرى و يکون قائماً علی خدمة هذا الامر الّذى به زلّت الاقدام يا طالب ان استمع نداء المظلوم تاللّه انّه ما اراد لک الّا ما يقرّبک الی اللّه رب الغيب و الاجهار ان اعمل بما وصّيناک من قبل بلسان الصّدق ثم تشبّث بذيل رحمة ربّک فالق الاصباح قل اى ربّ لک الحمد بما عرّفتنى و علّمتنى و اشهدتنى قد توجّهت بکلّى اليک و اسألک بان لا تدعنى بنفسى و انّک انت المقتدر المنّان و نذکر احبّائى فى السّين قل ان افرحوا بذکرى و ثنائى لعمر اللّه يبقى لکم ما جرى من هذا القلم الّذى شهد انّه لا اله الّا انا المقتدر العزيز الفضّال انّا نرا کم فى حبّ اللّه و امره و نوصيکم بالاستقامة الکبرى لانّ بها ترتفع اعلام النّصرة بين الارض و السّماء و يغرّد عندليب البقاء فى الاجواء انّه لا اله الّا هو المقتدر علی الاکوان خذوا رحيق البيان باسمى ثم اشربوا منه بذکرى الّذى احاط الجهات طوبى لکم بما تشرّفتم بذکر اللّه و توجّهتم فى يوم فيه زلّت الاقدام يا قلمى توجّه الی اهل الميم الّذين شربوا رحيق العرفان فى ايّام الرّحمن و فازوا بهذا الذّکر الجميل انّا سمعنا نداء کلّ واحد منکم و نرا کم علی ماانتم عليه فضلاً من لدى اللّه العليم الخبير انّا نذکر من سمّى بعلی فى ملکوت الاسماء ليسمع نداء ربّه الکريم ان يا علی ان استعدّ لاصغاء نداء ربّک الابهى الّذى ارتفع من هذا الافق الاعلى و المنظر الاسنى لتشهد بما شهد الملأ الاعلى انّ ربّک لهو المبيّن العزيز الحميد قم علی خدمة موليک علی شأن يتحيّر به ما سواک کذلک يأمرک قلم الامر من هذا المقام المنير تمسّک بالعروة الاستقامة و تشبّث بذيل رحمة ربّک و قل يا اله الاسماء و فاطر السّماء و المهيمن علی الاشياء اسألک باسمک الّذى به انکسر ظهر الاصنام بان تجعلنى ناطقاً بذکرک ذاکراً بين خلقک ثم ايّدنى علی خدمة امرک انّک انت المقتدر على ما تشاء لا اله الّا انت العليم الحکيم ان يا علی اذ ارتفع نعيق ناعق فى هناک دعه بنفسه و توجّه بقلبک الی افق ظهور ربّک العزيز الحميد کذلک جرى اسمک من لسان المظلوم و نزّل لک ما يبقى به اسمک بين السّموات و الارضين و نذکر من سمّى بالباء و الزّاء ثمّ الرّاء و الکاف ليأخذه جذب آيات ربّه على شأن يقوم و ينادى قل تاللّه قد ظهر المحبوب و اتى الرّحمن بعرشه العظيم قل يا قوم لا تضيّعوا امر اللّه بينکم دعوا ما عند کم و خذوا ما اتيکم من لدى اللّه ربّ العالمين هذا يوم لا تنفعکم خزائن العالم و لا اعانة الامم توکّلوا علی اللّه و توجّهوا الی افقه المنير کذلک يعلّمک ربّک لتذکره بين عباده و تکون من الرّاسخين توجّه وجه القدم الی ارض النّون و يذکر الّذين آمنوا باللّه ربّ ما کان و ما يکون يا محمّد ان افرح بما يذکرک الفرد الاحد من هذا المقام الاعلی لتقوم علی ذکر مولاک الّذى سجن فى سبيل اللّه المهيمن القيّوم قل تاللّه قد جعل اللّه السّجن قصراً من الياقوت ينطق فيه مالک الملکوت انّه لا اله الّا انا العزيز المحبوب کن مستقيماً علی حبّ مولاک لانّ النّاعق ينعق بين العباد دعه بنفسه و تمسّک بحبل اللّه ربّ الغيب و الشّهود کذلک جرى من قلم الرّحمن ماء الحيوان ان اشرب و قل لک الحمد يا اله العالم و لک الشّکر يا مالک الوجود ان يا محمود يذکرک ربّک من هذا المقام المحمود و يناديک بنداء لو يجد من الوجود عرفه ليدعنّ ما عند هم و يطيرنّ فى هواء محبة ربّک العزيز الودود اذا فزت بآيات اللّه ان احفظها ثمّ اقرئها فى اللّيالی و الايّام لعمر اللّه تستضيئ بها الافاق و تستنير بها القلوب ايّاک ان تحزنک شئونات العالم کن ناظراً فى کلّ الاحوال الی هذا الافق الّذى منه ينادى الاسم الاعظم الملک للّه مالک الملوک قل يا عباد الرّحمن هل بينکم من ذى سمع ليسمع نداء اللّه و هل بينکم من ذى بصر لينظر ما ظهر فى يوم الموعود قل انّه يکتب لمن اراد اجر لقائه کذلک قضى الامر فى لوح محفوظ کم من عبد تقرّب و يا فاز و کم من عبد فاز باللّقاء اذ کان فى مقام بعيد کذلک يعلّمکم اللّه فضلاً من عنده انّه لهو العليم الحکيم انّا نذکر الّذين آمنوا باللّه فى ارض زيّنت باسم الميم فى هذا اللّوح المنيع يا اهل الميم و الزاء ان افرحوا بذکر اللّه و توجّهوا بقلوب نوراء الی مشرق الطّور الّذى فيه ينادى مالک الظّهور الملک للّه الفرد الواحد العليم الخبير انّا نذکر الّذين شربوا رحيق الوحى و فازوا بعرفان اللّه فى هذا الفجر المنير طوبى لکم بما سمعتم و اقبلتم و آمنتم باللّه ربّ العالمين انتم فى مقاعد کم و لسان المظلوم يذکر کم من هذا المقرّ الّذى يطوفه الملأ الاعلى ثمّ اهل مدائن الاسماء و الّذين يطوفون حول عرش عظيم هنيئا لکم و مريئاً لکم يا اصفياء اللّه و احبّائه طوبى لوجوهکم بما توجّهت و لقلوبکم بما اقبلت و لنفوسکم بما طارت و لعيونکم بما رأت و لالسنکم بما نطقت بثناء اللّه الملک الحقّ المبين انّا نوصيکم بالاستقامه علی امر اللّه لانّها تنفعکم فى الدّنيا و الآخرة انّ ربّکم الرّحمن لهو العليم خذوا کأس الحيوان باسمى الرّحمن رغماً للّذين کفروا بالبرهان اذ ظهر من لدى اللّه العزيز الجميل ثمّ اشربوا منه تارةً باسمه و طوراً بذکرى البديع المنيع کذلک يذکّر کم من سجن فى سبيل اللّه و کذلک يذکّر کم من استقرّ علی هذا المقام الکريم البهاء عليکم من لدى اللّه مقصود العارفين و الحمد للّه ربّ العالمين .
