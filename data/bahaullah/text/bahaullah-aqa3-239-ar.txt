	بسم اللّه العليم
للّه علم السّموات و‌الارض و قد تجلّى باسمه العليم علی کلّ الاشياء. طوبى لمن اقبل اليه انّه مرأت علمه بين الخلايق و انّه لعليم خبير و‌انّ اوّل العلم هو عرفانه و الّذى فاز به فاز بکلّ العلوم و الّذى حُرم لن يصدق عليه اسم العلم ولو عنده علم الاوّلين و‌الآخرين لانّ علم الّذى هو‌المقصود هو‌عرفان الّذى ظهر بالحق و هو عرفان‌اللّه بين العالمين. طوبى لعالم فاز‌بالمعلوم و‌ترک‌الموهوم فى ايّام ربّه انّه من‌اعلی الخلق عند‌اللّه و‌له مقام عزّ‌کريم و الّذى قرء‌کلّ العلوم و ما فاز‌بالمعلوم انّه اجهل الجهلا ليس‌له ذکر فى لوح عزّ حفيظ. انّا وصّينا مظاهر العلم کلّهم اجمعين بان لايتوقّفوا فى هذا‌الامر و لو‌کان عندهم جواهر‌العلم و‌الحکمة کذلک ذکر‌الحق و قضى الامر‌من لدى‌اللّه ربّک و ربّ العالمين و اذا جاء‌الوعد و شقّت السماء بالسّحاب و‌اتى الموعود بسلطان مبين اعترضوا عليه عباد‌الّذينهم خلقوا بقوله و‌کفروا باللّه الملک العزيز العليم. قل انّ العلم مرآت الاحديّة تنطبع فيها تجليّات العلميّة من‌لدن عالم حکيم و‌منه تبيّض الوجوه و تطمئنّ القلوب و ‌تستنير صدور‌المخلصين و ‌يستبشر منه اهل ملأ الاعلی ثمّ عباد‌اللّه المريدين. طوبى لعالم يمشى علی وجه الارض و يدعوا النّاس الی‌اللّه الملک العزيز العظيم. تاللّه انّ‌الارض تباهى به و‌السّماء تفتخر بوجوده و يکبرنّ عليه کلّ الاشياء فى کلّ حين و منه تنتشر آثار‌اللّه بين عباده و تستقرّ ارکان الدّين بين الخلايق اجمعين. کلّ ذلک علم الّذى يهدى الناس الی‌اللّه ربّهم و ربّ آبائهم الاوّلين و‌لکن الّذى يدعوا‌النّاس الی‌الجهل و يمنعهم عن اللّه انّه کالثّعبان لسع اکباد النبييّن من قبل کما سمعت من قرون الاولی و شهدت فى هذا‌الظّهور الامنع البديع. فويلٌ لعالم منه بکى روح القدس و ضجّ اهل ملأ الاعلی و صاح من فى مدائن البقاء و ناحت ملائکة المقرّبين و منه اصفرّت وجوه الابرار و سفکت دماء الاحرار و نصبت رؤس الّذينهم آمنوا علی رماح المشرکين تاللّه لو ينقطع العلم عن الدّنيا لخير من ان يکون عالماً و ينکر امر اللّه حين ظهوره لانّ به احمرّت‌الارض من دماء المقرّبين. و لو‌يرتقى احد‌الی معارج القدس و يکشف اللّه عن بصره حجبات الهوى ليشهد اکباد الاصفيا مشبّکة من سهام العلماء الّذين اعترضوا علی مظاهر الامر فى کلّ الاعصار و باعتراضهم ورد عليهم ما سقطت اوراق سدرة المنتهى و‌اضطربت افئدة الاولياء و‌تذرّفت عيون العارفين. طوبى لعالم ما منعه العلم عن التوجّه الی‌اللّه العالم العليم و ويلٌ لعالم سلّ سيف العلم و حارب به مع اللّه ربّه و ربّ الاوّلين و‌الآخرين. و اذکر حين الّذى اتى علىّ بالحق علی ظلل البيان و انکره علماء الفرقان الی ان قتلوه بالظّلم هؤلاء الظّالمين بعد الّذى اتى بحجّة بها ثبت ايمانهم و ايمان آبائهم باللّه المقتدر العزيز القدير. و کذلک فانظر فى تلک الايّام انّ الّذى سمّى بمحمّد من اهل القاف غرّته العلم علی شأن کفر باللّه و اعرض عن آياته و‌کان قائد المشرکين الی اسفل الجحيم مع الّذى حضر تلقاء‌الوجه فى العراق و تجلّى عليه نيّر الآفاق بانوار الحکمة و المعانى و کان من الفائزين. و فى تلک الايّام اعترض علی الرّحمن و يوسوس فى صدور النّاس و اراد ان يدحض الحق بما عنده بعد الّذى لم يطّلع من العلوم الظاهرية الاّ علی‌قدر معلوم و بذلک احتجب عن الّذى عنده علم ما کان و ما يکون و مع انّه کان موقناً بانّ البيان ما نزّل الاّ لهذا الامر و ماتحرّک قلم اللّه الاّ علی هذا‌النّباء‌المحتوم. و‌کذلک فانظر فى الّذى سمّى باسمه من اهل الصّاد الّذى هاجر مع اللّه و ما کان مستطيعاً ان يتکلّم تلقاء احد من خدّامنا فلما وردنا فى السّجن ارتکب ما نهى عنه. اطردناه بالعدل قام علی النّفاق و فعل ما لافعله النّمرود و اتّحد مع الّذى ربّيناه فى سنين معدودات ثمّ اتّحد مع اعداء‌اللّه و فى‌کلّ يوم مکرا مکراً لتخمد به نار اللّه الّتى يسمع من زفيرها بانّه لا اله الاّ هو المهيمن القيّوم. کذلک قصصنا لک ماورد علی هذا الغلام المسجون بعد الّذى ما دعونا الکلّ الاّ الی‌اللّه و حفظناهم عن ضرّ الّذينهم قاموا علی الاعراض فى يوم اضطربت فيه افئدة النّاس و تحيّرت فيه العقول. طوب لک يا ايّها النّاظر الی الوجه بما آمنت باللّه و آياته و سمعت ندائه و کنت من الّذين الی شطر الفضل هم سارعون و ما منعک ضوضاء المشرکين عن التّوجّه الی‌اللّه العزيز المحبوب. اذاً قم علی الامر ثمّ بلّغ النّاس ذکر ربّک لعلّ بذلک يقومنّ الرّاقدون لاتحزن عن شئ و توکّل علی‌اللّه الّذى خلقک و سوّاک و ايّدک علی عرفان نفسه بعد الّذى کلّ عنه معرضون الاّ من شرب رحيق الابهى باسم ربّه العلىّ الاعلی و انقطع عما خلق فى الارض و السّماء و‌ما منعه ضوضاء الاشقياء عن التوجّه الی شطر‌اللّه المقدّس العزيز الودود. ان اجهد لئلاّ يفوت منک ما کنت مستطيعاً به فى نصرة مولاک و هذا خير الاعمال عند‌اللّه الّذى يسجد له من فى ملکوت الغيب و الشّهود. و البهاء‌عليک و علی الذّين يسمعون قولک فى ذکر اللّه المقتدر العزيز القيّوم.
