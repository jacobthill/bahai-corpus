	هو العزيز الفرد القيّوم
بسم اللّه الرّحمن الرّحيم
سبحان الّذى يذکر من يشآء بامر من عنده و يلقى من يشاء ما يريد من علمه لا اله الّا هو العلّى العظيم بيده ملکوت کلّ شىٌ و يعطى عباده ما يغنيه عن العالمين يحيى و يميت ثمّ يميت و يحيى و انّه لهو الفرد المتعالی العزيز الجميل قل انّ فى تنزيل البلايا و المحن لآيات للمؤمنين و فى جريان الدّموع علی الخّد و سفک الدّماء لظهورات للعاشقين ثمّ اعلم بانّا قطعنا السّبيل حتّى وردنا فى شاطئ بحر عظيم و هذا ما ذکرناه فى سنة القبل ان انتم من العالمين ثمّ استوينا علی الفلک و هذا ما قدّر من لدى اللّه المقتدر الحکيم و سکن البحر عن الامواج حتّى وصلوا امانات اللّه فى ساحل البحر و خرجنا عنها باذن من حميد عليم و وردنا فى شهر الّذى کان مذکورا عند کم و مکثنا فيه عدّة من الشهور بما قدّر فى الواح قدس حفيظ الی ان قضى الوعد فيه و تمّت ميقاته خرجنا الی ارض اخرى خلف جبال متين و هذا ما سئلنا اللّه فى مُذ الايام بان يخرجنا عن بين هؤلاء و يحول بينى و بين المنافقين ما ينقطع عنّى ايدى المبغضين و لذا اسکنّا خلف الجبال فى مقرّ الّذى لن يرفع عنّا الضّجيج و لو يرفع لن يسمعه آذان المشرکين و رفعت النعمة عن بينهم و رجعت الی کنز اللّه المتعالی القدير و لکنّ المؤمنون يسمعون ندائنا و يصل اليهم نفحات الامر فى کلّ حين قل يا قوم لا تفرحوا بما غابت الشمس خلف سحاب امر غليظ و لا تطمئنّوا بالدّنيا و زخرفها فسوف يأخذ اللّه عنکم کلّها تفتخرون به اليوم و تکوننّ به لمن المستکبرين قل فو اللّه لا يُعطيکم بدخشان القدره ما فقد عنکم من هذا اللّعل الدّرى المنير و لا يهبکم سحاب العزّة ما منع عنکم فى غيبتى ان انتم من العارفين و لن تجدوا فى صدف البحر و الامکان ما فقد عنکم من هذا اللّؤلؤ الحفيظ و لا تبذل عليکم اشجار الدّهر بمثل ما غاب عنکم من هذا الثّمر الرّطب الجنىّ اللّطيف قل هل تقدرون بان تسدّوا نسمات اللّه اوتحبسوا روايح العزّة او تمنعوا النّاس عن صراط اللّه الحکم العدل المحيط لا فو ربّ العزّة لن تقدروا و لن تستطيعوا و لو يؤيّد کم کلّ من فى السّموات و الارضين قل انّ الفرعون و ملأه اجتمعوا بان يسدّو ابواب الفضل و يمنعوا النّاس عن رحمة اللّه و مکروا فى ذلک بکلّ ما کانوا مستطيعاً عليه فى زمان بعيد فلمّا جاء الوعد فتح اللّه باب العناية فى بيته و جرى فيها بحر الرّحمة و منع الفرعون عن قطرة منه رغماً لانفه و کذلک کان الامر من قبل من لدن سلطان آمر حکيم و کان الفرعون يتفحّص عن موسى فى اطراف الارض و هو فى بيته و ما کان من الشّاعرين کذلک نقصّ عليک من قصص الحق لتقرّ بها عيناک و تفرح بها و تکون من الموقنين و تعلم بانّ الأمر بيده و الحکم فى قبضته اقتداره و السّلطان فى کفّ ارادته يفعل ما يشاء و لا يمنعه مکر ماکر و لا تدبير مدبّر و لا اعراض معرض و لا انکار کلّ شيطان مريد فاطمئن فى نفسک ثمّ امش علی اثرى و لا تجاوز عنه و لو يجتمع عليک کلّ معرض شقىّ فامش علی قدمى و لو يخالفک فى ذلک عينک فاقلعها و لا تلتفت اليها ان کنت من السّامعين فاحبب من احبّ اللّه و يطابق فعله قوله و يصدّقه ارکانه ثم اعرض عن کلّ مکّار لئيم ثمّ وصّ العباد بان لا يفسدوا فى ارض اللّه و لا يختلفوا فى امره و لا يعقّبوا کلّ مريد اثيم ايّاک ان لا تمش فى الارض الّا بالحکمة و لا تکلّم الّا بالحکمة و کن من المتفرّسين لا تفکّ فمک ما لن تجد اذن سميع و ان وجدت اذن و اعية فالق ما القيک اللّه بجوده ان کنت من الذّاکرين و لا تفتح عيناک الّا الی وجه جميل و ان وجدت بصر حديد فاشهده ما ستر فيک من جمال العلم لتکون من المبشّرين و لا تضيّع بذر الحکمة فى اراضى الجرزة و لا تکن من المسرفين و ان وجدت ارض طيّبة اودع فيها حَبّ الحکمة و العرفان لينبت منها نبات حسن بديع و لا تمطر علی هياکل الفانية امطار الباقية فامطر علی اجساد الّتى لو تسقى قطرة منه لتحيى باذن اللّه الملک العزيز الکريم فاجهد يا اخى فى کلمة اللّه و لا تضيّعها و لا تنشرها بين يدى الغافلين لاّن الناس فى علق غير مخلّقة لن يقدرنّ ان يصلنّ الی هواء القدس و يدخلنّ فى رفرف عزّ مکين لانّهم قطعوا جناحهم بايديهم و بذلک منعوا عن سرادق الخلد و وقعوا فى ارض الذلّة مع حسرت عظيم و منعوا عن آذانهم نغمات اللّه و عن عيونهم جمال اللّه الفرد العليم الحکيم ثم ذکّر من لدنّا کلّ صغير و کبير من الّذين تجد منهم روايح الصّدق و من سيما هم نضرة النّعيم ثمّ استر الامر عن الّذين وجوههم اليک و قلوبهم الی اعدائک و يمکرون فى کلّ حين فاعرض عن هؤلاء و لا تأنس معهم و لا تجلس فى غرفهم و لا تسکن فى مقاعد هم و کن فى ستر عظيم تجنّب عن مثل هؤلاء و لا تفرّط فى جنب اللّه و لا تتجاوز عمّا امرت به و لا تکن من المتجاوزين ثمّ ذکرّ من لدنّا احباب اللّه الذّين تفرّقوا فى اقطار العالم من کلّ شطر بعيد ثمّ فى ارض الالف حرف الفاء و فى ارض الياء حرف الرّاء و الضّاد و بلّغهم ما ارسلناه اليک ليکوننّ من الذّاکرين ليفرح بذلک قلوبهم و يجدّد ارکانهم و يجدوا روايح القميص عن هذا اللّوح المنيع ثمّ الّذين کانوا فى ارض الشّين و الصّاد و من دونهما مداين اللّه العزيز الغالب القدير لعلّ يشدّون ظهر هم لنصر اللّه و يبلّغوا النّاس ما بلّغهم اللّه و انّه لجواد کريم و کذلک القيک من کلّ حکمة ما قدّر اللّه لک لتستأنس بها و تکون من الشّاکرين ثمّ ذکّر الذّينهم کانوا فى حولک و بشّر هم بذکرنا ايّاهم ليسّروا فى انفسهم و يکوننّ من المنقطعين ثمّ ذکّر من لدنّا حرف الجيم و کبّر عليه و علی ابنه و علی الّذينهم کانوا فى حول البيت لطائفين ثمّ ارع البيت و ساکنيها بکلّ ما ينبغي لک الی ان يظهر اللّه ما اراد بامره و انّه لهو العزيز القادر الرّفيع و انّ ما وعدناک حين الحضور بان نرسل اليک ما هو المکنون فى اصداف القدرة من لئالی قدس ثمين ما ارسلنا الی حينئذ و سأرسل من بعد اذا شاء اللّه و انّه يوفّى اجور المحسنين و کذلک القيناک قول الحقّ و انزلنا اليک هذا اللّوح حين الّذى غطّى الوجه من ماء بيض منير و السّلام عليک و على من معک و على من سمع نصح هذا العبد من هذا القلم البديع ١٥٢
